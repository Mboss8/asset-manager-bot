import type { Context } from "telegraf";
import { db, groupsTable } from "@workspace/db";
import { eq } from "drizzle-orm";
import { editOrSend, buildKeyboard, writeAudit } from "../helpers.js";
import { invalidateGroupsCache, listEnabledGroups } from "../group-service.js";
import { getUserByTelegramId } from "../user-service.js";

function formatChatType(t: string | null | undefined): string {
  if (t === "group") return "群";
  if (t === "supergroup") return "超级群";
  if (t === "channel") return "频道";
  return t ?? "unknown";
}

function formatId(id: number | null | undefined): string {
  if (!id) return "未设置";
  return `<code>${id}</code>`;
}

export async function showGroupsPanel(ctx: Context, telegramId: string): Promise<void> {
  const me = await getUserByTelegramId(telegramId);
  if (!me) {
    await ctx.reply("❌ 用户不存在");
    return;
  }

  const groups = await db.select().from(groupsTable).orderBy(groupsTable.id);

  let text = `<b>📡 群组路由中心</b>\n<i>让播报有方向，让权限有边界。</i>\n\n`;

  if (groups.length === 0) {
    text += `⚠️ 当前 groups 表为空。\n\n你可以先把机器人拉进群/频道，然后用 /chatid 获取 chat id，再添加记录。\n`;
  } else {
    for (const g of groups) {
      const enabled = g.isEnabled === 1 ? "✅ 启用" : "⛔ 禁用";
      const deleted = g.deletedAt ? "🗑 已软删" : "";
      text += `#${g.id} ${enabled} ${deleted}\n`;
      text += `• tgChatId: <code>${g.tgChatId}</code>\n`;
      text += `• 类型: ${formatChatType(g.chatType)}\n`;
      text += `• 默认频道: ${formatId(g.defaultReportChannelId)}\n`;
      text += `• 财务频道: ${formatId(g.financeReportChannelId)}\n`;
      text += `\n`;
    }
  }

  text += `<b>操作</b>\n`;
  text += `• 选择一条记录可启用/禁用\n`;
  text += `• 可设置默认频道 / 财务频道\n`;
  text += `• 所有改动即时影响 dispatchBroadcast（30s 缓存会自动刷新）`;

  const buttons = groups.map((g) => ({
    text: `${g.isEnabled === 1 ? "✅" : "⛔"} #${g.id}`,
    callback_data: `GROUPS:OPEN:${g.id}`,
  }));

  const footer = [
    { text: "🔄 刷新缓存", callback_data: "GROUPS:CACHE:INVALIDATE" },
    { text: "🔙 返回系统设置", callback_data: "M:SET" },
    { text: "🏠 返回主页", callback_data: "M:HOME" },
  ];

  const keyboard = buildKeyboard(buttons, 3, footer);
  await editOrSend(ctx, text, keyboard);
}

export async function showGroupCard(ctx: Context, groupId: number, telegramId: string): Promise<void> {
  const me = await getUserByTelegramId(telegramId);
  if (!me) {
    await ctx.reply("❌ 用户不存在");
    return;
  }

  const rows = await db.select().from(groupsTable).where(eq(groupsTable.id, groupId)).limit(1);
  const g = rows[0];
  if (!g) {
    await ctx.reply("⚠️ 群组记录不存在");
    return;
  }

  const text =
    `<b>📡 群组配置详情</b>\n\n` +
    `ID: <code>${g.id}</code>\n` +
    `tgChatId: <code>${g.tgChatId}</code>\n` +
    `类型: ${formatChatType(g.chatType)}\n` +
    `启用: ${g.isEnabled === 1 ? "✅" : "⛔"}\n` +
    `默认频道: ${formatId(g.defaultReportChannelId)}\n` +
    `财务频道: ${formatId(g.financeReportChannelId)}\n\n` +
    `<i>提示：频道 ID 通常是负数。</i>`;

  const buttons = [
    {
      text: g.isEnabled === 1 ? "⛔ 禁用" : "✅ 启用",
      callback_data: `GROUPS:TOGGLE:${g.id}`,
    },
    { text: "📌 设置默认频道", callback_data: `GROUPS:SETDEF:${g.id}` },
    { text: "💰 设置财务频道", callback_data: `GROUPS:SETFIN:${g.id}` },
  ];

  const footer = [
    { text: "🔙 返回列表", callback_data: "GROUPS:LIST" },
    { text: "🏠 返回主页", callback_data: "M:HOME" },
  ];

  const keyboard = buildKeyboard(buttons, 2, footer);
  await editOrSend(ctx, text, keyboard);
}

export async function toggleGroupEnabled(ctx: Context, groupId: number, telegramId: string): Promise<void> {
  const me = await getUserByTelegramId(telegramId);
  if (!me) {
    await ctx.answerCbQuery("❌ 用户不存在", { show_alert: true });
    return;
  }

  const rows = await db.select().from(groupsTable).where(eq(groupsTable.id, groupId)).limit(1);
  const g = rows[0];
  if (!g) {
    await ctx.answerCbQuery("⚠️ 群组不存在", { show_alert: true });
    return;
  }

  const next = g.isEnabled === 1 ? 0 : 1;

  await db.update(groupsTable).set({ isEnabled: next }).where(eq(groupsTable.id, groupId));
  invalidateGroupsCache();

  await writeAudit(
    me.id,
    next === 1 ? "GROUP_ENABLE" : "GROUP_DISABLE",
    "groups",
    groupId,
    `tgChatId=${g.tgChatId}`,
    "MEDIUM",
  );

  await ctx.answerCbQuery(next === 1 ? "✅ 已启用" : "⛔ 已禁用");
  await showGroupCard(ctx, groupId, telegramId);
}

export async function invalidateCache(ctx: Context, telegramId: string): Promise<void> {
  const me = await getUserByTelegramId(telegramId);
  if (!me) {
    await ctx.answerCbQuery("❌ 用户不存在", { show_alert: true });
    return;
  }

  invalidateGroupsCache();
  const enabled = await listEnabledGroups();

  await writeAudit(
    me.id,
    "GROUP_CACHE_INVALIDATE",
    "routing",
    null,
    `enabledGroups=${enabled.length}`,
    "LOW",
  );

  await ctx.answerCbQuery("🔄 缓存已刷新");
  await showGroupsPanel(ctx, telegramId);
}
